        async function processFile() {
            try {
                showStatus('Processando arquivo...', 'info');
                
                // Usar o arquivo j√° enviado ou o arquivo selecionado pelo usu√°rio
                let file;
                const fileInput = document.getElementById('fileInput');
                
                if (fileInput.files && fileInput.files[0]) {
                    file = fileInput.files[0];
                } else {
                    // Usar o arquivo j√° carregado
                    file = await window.fs.readFile('Aloca√ß√µes 052025.xlsx');
                }

                const arrayBuffer = file instanceof ArrayBuffer ? file : await file.arrayBuffer();
                const workbook = XLSX.read(arrayBuffer);
                const worksheet = workbook.Sheets["ALOCA√á√ÉO MAIO"];
                const jsonData = XLSX.utils.sheet_to_json(worksheet, { header: 1 });

                // Aplicar corre√ß√µes
                let correcoesAplicadas = 0;
                const logCorrecoes = [];

                for (let i = 2; i < jsonData.length; i++) {
                    if (jsonData[i] && jsonData[i][1]) {
                        const nomeAtual = jsonData[i][1].toString().trim();
                        
                        if (correcoesNomes[nomeAtual]) {
                            logCorrecoes.push({
                                linha: i + 1,
                                original: nomeAtual,
                                corrigido: correcoesNomes[nomeAtual]
                            });
                            jsonData[i][1] = correcoesNomes[nomeAtual];
                            correcoesAplicadas++;
                        }
                    }
                }

                // Criar novo workbook com dados corrigidos
                correctedWorkbook = XLSX.utils.book_new();
                const novaWorksheet = XLSX.utils.aoa_to_sheet(jsonData);
                XLSX.utils.book_append_sheet(correctedWorkbook, novaWorksheet, "ALOCA√á√ÉO MAIO");

                // Mostrar log de corre√ß√µes
                displayCorrections(logCorrecoes);
                
                // Atualizar estat√≠sticas
                document.getElementById('corrections').textContent = correcoesAplicadas;

                // Mostrar elementos de download
                document.getElementById('downloadBtn').style.display = 'inline-block';
                document.getElementById('downloadSection').style.display = 'block';
                
                showStatus(`Processamento conclu√≠do! ${correcoesAplicadas} corre√ß√µes aplicadas.`, 'success');

            } catch (error) {
                console.error('Erro ao processar arquivo:', error);
                showStatus('Erro ao processar arquivo: ' + error.message, 'error');
            }
        }

        function displayCorrections(corrections) {
            const correctionsList = document.getElementById('correctionsList');
            const correctionsLog = document.getElementById('correctionsLog');
            
            correctionsList.innerHTML = '';
            corrections.forEach(correction => {
                const div = document.createElement('div');
                div.className = 'correction-item';
                div.innerHTML = `
                    <strong>Linha ${correction.linha}:</strong><br>
                    <span class="original-name">‚ùå ${correction.original}</span><br>
                    <span class="corrected-name">‚úÖ ${correction.corrigido}</span>
                `;
                correctionsList.appendChild(div);
            });
            
            correctionsLog.style.display = 'block';
        }

        function downloadCorrectedFile() {
            if (!correctedWorkbook) {
                showStatus('Nenhum arquivo processado para download.', 'error');
                return;
            }

            try {
                const wbout = XLSX.write(correctedWorkbook, { bookType: 'xlsx', type: 'array' });
                const blob = new Blob([wbout], { type: 'application/octet-stream' });
                
                const url = URL.createObjectURL(blob);
                const a = document.createElement('a');
                a.href = url;
                a.download = 'Aloca√ß√µes_052025_Corrigido.xlsx';
                document.body.appendChild(a);
                a.click();
                document.body.removeChild(a);
                URL.revokeObjectURL(url);
                
                showStatus('Arquivo baixado com sucesso!', 'success');
            } catch (error) {
                console.error('Erro ao baixar arquivo:', error);
                showStatus('Erro ao baixar arquivo: ' + error.message, 'error');
            }
        }

        function showStatus(message, type) {
            const status = document.getElementById('status');
            status.textContent = message;
            status.className = type === 'success' ? 'status-success' : 
                              type === 'error' ? 'status-error' : '';
            status.style.display = 'block';
            
            if (type === 'success') {
                setTimeout(() => {
                    status.style.display = 'none';
                }, 5000);
            }
        }
